Массивы в JavaScript

Массивы. Синтаксис

Массив – это особый подвид объектов, предназначенный для хранения упорядоченных коллекций данных.

Cинтаксис для создания пустого массива: let arr = [элемент-1, элемент-2, ...элемент-n];

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',];

Элементы массива нумеруются начиная с нуля. Мы можем получить элемент, указав его номер в квадратных скобках []:

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; console.log(cars); // - выведет ['Мазда', 'БМВ', 'Мерседес', 'Лада'] console.log (cars[0]); // - получим Мазда console.log (cars[1]); // - получим БМВ Мы можем заменить элемент в массиве, указав в квадратных скобках [] его индекс: cars[2] = 'Порше'; МЫ можем добавить новый элемент к существующему массиву, указав несуществующий индекс. cars[4] = 'Ферарри'; console.log(cars); // - массив теперь: ['Мазда', 'БМВ', 'Порше', 'Лада', 'Ферарри']

Общее число элементов массива содержится в его свойстве length:

console.log(cars); // - массив из 5 элементов: ['Мазда', 'БМВ', 'Порше', 'Лада', 'Ферарри'] console.log(cars.length); // - получим 5

Cамый простой способ очистить массив – указать, что длина массива равна нулю: arr.length = 0

Массивы могут содержать элементы, которые тоже являются массивами, называется (многомерный массив).

let arr = [[1, 2, 3], ['a', 'b', 'c']]; console.log(arr [1][2]); // - получим c

В массиве могут храниться элементы любого типа.

let arr = [ 'Джон', { address : 'Москва, ул. Ленина, д.32, кв. 77', } 31, function() { alert(`привет, ${arr[0]}`) } ]; alert (arr[1]. address ); // - получить значение свойства объекта, который хранится под индексом 1 arr[3](); // - получить элемент с индексом 3, который является функцией и выполнить её.

Чтобы получить последний элемент массива нужно использовать метод arr.at( -1).

Массив копируется по ссылке также, как и объект.

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; let auto = cars; auto[3] = 'Волга'; console.log (cars[3]); // - вернёт 'Волга' console.log (auto[3]); // - вернёт 'Волга'

Методы массивов

Методы pop / push, shift / unshift

Очередь – упорядоченная коллекция элементов, один из самых распространённых вариантов применения массива.

Массивы поддерживают два типа доступа к элементам: очередь - работает с началом массива и стек - работает с концом массива.

Работать с концом массива быстрее, чем с началом массива.

Методы, работающие с концом массива:

arr.pop( ) - удаляет последний элемент из массива и возвращает его:

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; console.log (cars.pop()); // - удаляем Лада и выводим его console.log(cars); // - получим ['Мазда', 'БМВ', 'Мерседес',] И cars.pop() и cars.at (-1) возвращают последний элемент массива. Но cars.pop() также изменяет массив, удаляя его последний элемент.

arr.push( ) - добавляет элемент в конец массива:

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; cars.push ('Рено'); // - добавили элемент в конец массива console.log(cars); // - массив изменился: ['Мазда', 'БМВ', 'Мерседес', 'Лада', 'Рено'];

Методы, работающие с началом массива:

arr.shift( ) - удаляет из массива первый элемент и возвращает его:

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; console.log (cars.shift()); // - удаляем первый элемент массива и возвращаем его 'Мазда' console.log (cars); // - массив изменился, теперь ['БМВ', 'Мерседес', 'Лада']

arr.unshift( ) - добавляет элемент в начало массива:

let cars = ['Мазда', 'БМВ', 'Мерседес', 'Лада',]; cars.unshift ('Ламборджини'); console.log(cars); // - массив изменился, теперь ['Ламборджини', 'Мазда', 'БМВ', 'Мерседес', 'Лада']

Методы push() и unshift() могут добавлять сразу несколько элементов:

let cars = ['Лада']; cars.unshift ('Порше', 'БМВ'); cars.push ('Мазда', 'Ниссан'); console.log(cars); // - добавили элементы в начало и конец массива, теперь ['Порше', 'БМВ', 'Лада', 'Мазда', 'Ниссан']

Добавлениe / удалениe / копирование

splice

Метод arr.splice (args...) умеет всё: добавлять, удалять и заменять элементы массива. Его синтаксис:

arr.splice (начальный индекс, количество удаляемых элементов, elem-1, elem-2, ... elem-n); // - начиная с позиции начальный индекс, удаляем количество элементов и вставляем elem-1, elem-2, ... elem-n вместо них. splice() также возвращает массив из удалённых элементов.

Удаление определенного элемента массива:

let cars = ['Порше', 'Майбах', 'Митсубиси',]; cars.splice (1, 1); // - начиная с позиции 1, удалить 1 элемент console.log(cars); // - массив изменился, теперь ['Порше', 'Митсубиси']
Удаление нескольких элементов массива и замена их другими элементами:
let cars = ['Порше', 'Майбах', 'Митсубиси',]; cars.splice (0, 2, 'Лада', 'Тойота', 'Ниссан', 'Хендай'); // - начиная с позиции 0, удалить 2 элемента и заменить их другими элементами console.log(cars); // - массив изменился, теперь ['Лада', 'Тойота', 'Ниссан', 'Хендай', 'Митсубиси']; let deleteElements = cars.splice (0, 2); // - создаем новый массив с удаленными элементами массива cars console.log(deleteElements); // - смотрим на новый массив из ранее удаленных элементов: выведет ['Лада', 'Тойота'] console.log(cars); // - массив изменился, теперь ['Ниссан', 'Хендай', 'Митсубиси']

Добавление элементов в массив без удаления

let text = ['Я', 'люблю', 'JavaScript',]; text.splice (1, 0, 'не'); // - начиная с позиции 1, удалить 0 элементов, добавить элемент 'не' после позиции 1 text.splice (4, 0, 'он', 'сложный'); // - начиная с позиции 4, удалить 0 элементов, добавить 'он', 'сложный' после позиции 4 console.log(text); // - массив изменился, теперь ['Я', 'не', 'люблю', 'JavaScript', 'он', 'сложный'];

В этом и в других методах массива допускается использование отрицательного индекса, что позволит начать отсчёт элементов с конца.

slice, concat

Метод slice( ) возвращает новый массив, в который копирует элементы, начиная с индекса start и до (не включая) end.

let footballClubs = ['Манчестер Сити', 'Ливерпуль', 'Челси', 'Арсенал',]; console.log (footballClubs.slice (1, 3)); // - вернет новый массив ['Ливерпуль', 'Челси']

Оба индекса start и end могут быть отрицательными. В таком случае отсчёт будет осуществляться с конца массива.

Можно вызвать slice( ) и вообще без аргументов: arr.slice() - создаcт копию массива arr.

Метод arr.concat( ) создаёт новый массив, в который копирует данные из других массивов и дополнительные значения. Он принимает любое количество аргументов, которые могут быть как массивами, так и простыми значениями.

let planets = ['Земля', 'Марс', 'Сатурн',]; создали копию массива planets и добавили в этот массив новые элементы, которые передали как аргументы в метода concat() let allPlanets = planets.concat ('Венера', 'Юпитер', 'Нептун',); console.log(allPlanets); результат: ['Земля', 'Марс', 'Сатурн', 'Венера', 'Юпитер', 'Нептун']

Перебор элементов

Перебор: for..of

Одним из самых старых способов перебора элементов массива является цикл for по цифровым индексам:

let cars = ['БВМ', 'Порше', 'Мерседес', 'Ауди',]; for (let i = 0; i < cars.length; i++) { console.log (cars[i]); }

Но для массивов возможен и другой вариант цикла, for..of:

let cars = ['БВМ', 'Порше', 'Мерседес', 'Ауди']; for (let car of cars) { console.log(car); }

Цикл for..of не предоставляет доступа к номеру текущего элемента, только к его значению, но в большинстве случаев этого достаточно.

Перебор: forEach

Метод arr.forEach( ) позволяет запускать функцию для каждого элемента массива. Его синтаксис:

arr.forEach (function(элемент, индекс, массив) { // ... делать что-то с элементом }); let games = ['Футбол', 'Баскетбол', 'Волейбол', 'Теннис',]; // - forEach() выведет в консоль каждый элемент массива и покажет свою позицию: games.forEach ((item, index, arr) => { console.log (`${item} имеет позицию: ${index} в массиве: ${arr}`); });

Результат функции (если она вообще что-то возвращает) отбрасывается и игнорируется.

Поиск элементов

indexOf / lastIndexOf и includes

arr.indexOf (item, from) ищет item, начиная с индекса from, и возвращает индекс, на котором был найден искомый элемент, в противном случае -1.

arr.lastIndexOf (item, from) – то же самое, но ищет справа налево.

arr.includes (item, from) – ищет item, начиная с индекса from, и возвращает true, если поиск успешен.

let cars = ['Порше', 'Мерседес', 'БМВ', 'Опель',]; console.log (cars.indexOf ('Опель')); // - пытаемся найти указанный элемент, результат вернёт позицию элемента в массиве - 3 console.log (cars.includes ('БВМ')); // - вернёт true (указанный элемент есть в массиве)

Если мы хотим проверить наличие элемента, и нет необходимости знать его точный индекс, тогда предпочтительным является arr.includes( ).

find / findIndex / filter

Метод arr.find( ) используют для поиска в массиве из объектов с определенным условием. Принцип его работы:

let result = arr.find (function(item, index, array) { - Запускается цикл (перебор всех элементов). Функция вызывается по очереди для каждого элемента массива: item – очередной элемент index – его индекс array – сам массив - если искомый элемент соответствует условию поиска, то возвращается текущий элемент (item) и перебор прерывается - если все итерации оказались ложными, возвращается undefined }); Например, у нас есть массив пользователей, каждый из которых имеет поля id и name. Попробуем найти того, кто с id == 1: let users = [ {id: 1, name: 'Вася'}, {id: 2, name: 'Петя'}, {id: 3, name: 'Маша'} ]; let user = users.find (item => item.id == 1); console.log (user.name); // - получим только первый найденный элемент, который соответствует заданному условию поиска - Вася. Если бы элементов соответствующих условию было несколько, всё равно вернулся бы только один элемент - самый первый Аналогичная запись: let user = users.find (function(item, index, array) { return item.name === 'Петя'; }); console.log(user); // - получили в результат элемент массива - объект, соответствующий условию поиска {id: 2, name: 'Петя'}

Метод arr.findIndex( ) – по сути, то же самое, но возвращает индекс (позицию в массиве), на котором был найден искомый элемент, а не сам элемент, если ничего не найдено, то вернёт -1.

Метод find( ) ищет один элемент (первый соответствующий условию поиска), на котором функция-колбэк вернёт true.

Метод arr.filter( ) используется когда нужно отфильтровать массив. Принимает в качестве параметра функцию-фильтр. Она получает информацию о текущем элементе, проводится проверка и возвращается true или false. Если для элемента вернется true — элемент будет добавлен в результирующий массив, если false — исключен из него.

На тот случай, если найденных элементов может быть много, предусмотрен метод arr.filter( function(item, index, arr)). Метод filter( ) возвращает новый массив из всех подходящих элементов. Пример работы:

let users = [ {id: 1, name: 'Вася'}, {id: 2, name: 'Петя'}, {id: 3, name: 'Маша'}, ]; let someUsers = users.filter( item => item.id < 3); // - вернёт массив, состоящий из двух первых пользователей console.log (someUsers.length); // - проверили работу метода, длина массива равна 2 let userInfo = [ {name: 'Петя', age: 18, gender: 'мужчина',}, {name: 'Оля', age: 25, gender: 'женщина',}, {name: 'Артём', age: 30, gender: 'мужчина',}, {name: 'Саша', age: 35, gender: 'женщина',}, ] let findUser = userInfo.filter (item => item.gender === 'женщина'); console.log(findUser); // - вернёт новый массив, содержащий в себе элементы - объекты, которые соответствуют условию

Преобразование элементов

sort

Вызов arr.sort( ) сортирует на месте массив, меняя в нём порядок элементов.

let names = [ 'Максим', 'Руслан', 'Андрей', 'Евгений',]; names.sort( ); // - метод sort отсортирует элементы массива names по алфавиту console.log( names ); // - изменит порядок элементов в массиве, теперь: ['Андрей', 'Евгений', 'Максим', 'Руслан']

Числа сортируются по умолчанию как строки, поэтому привычный порядок может быть нарушен. Чтобы использовать собственный порядок сортировки, нужно предоставить методу sort() в качестве аргумента функцию, определяющую порядок, которая знает как сравнивать его элементы.

let numbers = [ 1, 2, 15, 7,]; numbers.sort(); // - отсортирует числа в массиве, как строки, что нам не подходит, т.к. порядок элементов будет нарушен, результат: [1, 15, 2, 7] numbers.sort (function(a, b) { return a - b; }); Чтобы работал нужный нам порядок сортировки чисел, передаём методу sort( ) функцию в качестве аргумента. sort(функция сортировки) аналогичная запись: numbers.sort ((a, b) => a - b); // - переписали на стрелочную функцию console.log(numbers); // - теперь привычный порядок элементов в numbers. [1, 2, 7, 15]

reverse

Метод arr.reverse( ) меняет порядок элементов в массиве на обратный.

let arr = [1, 2, 3, 4, 5]; arr.reverse(); console.log(arr); // - изменит на обратный порядок элементов в массиве. [5, 4, 3, 2, 1]

map

Метод arr.map( ) вызывает функцию для каждого элемента массива и возвращает новый массив с результатами выполнения этой функции для каждого элемента исходного массива. Метод arr.map( ) используют для трансформации массива.

arr.forEach( ) перебирает элементы; arr.map( ) трансформирует элементы и возвращает новый массив.

let result = arr.map (function(item, index, array) { // - возвращается новое значение вместо элемента }); Например, здесь мы преобразуем каждый элемент в его длину: let lengths = ['Bilbo', 'Gandalf', 'Nazgul']. map(item => item.length); console.log(lengths); // - получим [5, 7, 6]

split / join

Метод string.split (разделитель) разбивает строку на массив по заданному разделителю.

В примере ниже разделителем является строка из запятой и пробела. let string = 'Вася, Петя, Маша'; let names = string.split (', '); console.log(names); // - получим массив из строки. Результат: ['Вася', 'Петя', 'Маша']

У метода split( ) есть необязательный второй числовой аргумент – ограничение на количество элементов в массиве. Если элементов больше чем указано, то остаток массива будет отброшен.

let arr = 'Вася, Петя, Маша, Саша'.split(', ', 2); console.log(arr); // - получим всего 2 элемента в массив, результат: ['Вася', 'Петя']

Разбивка по буквам. Вызов split('') с пустым аргументом разбивает строку на массив букв:

let str = 'тест'; console.log (str.split ('')); // - получим массив букв ['т', 'е', 'с', 'т']

Вызов arr.join ('склеивающий элемент') делает в точности противоположное split( ). Метод join() создаёт строку из элементов массива, вставляя склеивающий элемент между элементами.

let arr = ['Вася', 'Петя', 'Маша',]; let str = arr.join ('; '); // - получим строку из элементов массива, вставляя между ними склеивающий элемент ( ; ) console.log(str); // - результат работы метода join(): 'Вася; Петя; Маша'

arr.reduce( ) - вычисляет одно единое значение на основе всего массива.

let numberList = [10, 20, 30, 40, 50]; // - создаем массив из чисел. метод reduce вычисляет одно значение на основе всего массива, вызывая функцию для каждого элемента и передаёт промежуточный результат между вызовами. let numSumResult = numberList.reduce (function(akk, item, index, array) { return akk += item; }, 0); // - указываем начальное значение, если не указано, то начальным значением будет первый элемент массива. console.log(numSummResult); // - результат: единое значение 150 (сумма всех элементов массива).

console.log (Array.isArray({})); // - вернёт false, указали пустой объект. console.log (Array.isArray([])); // - вернёт true, указали пустой массив.

!!! методы sort( ), reverse( ), splice( ) изменяют исходный массив.

Варианты неправильного применения массива:

  • Добавление нечислового свойства, например: arr.test = 5.
  • Создание «дыр», например: добавление arr[0], затем arr[1000] - (между ними ничего нет).
  • Заполнение массива в обратном порядке, например: arr[1000], arr[999] и т.д.
  • Массивы тщательно настроены в движках JavaScript для работы с однотипными упорядоченными данными, поэтому, используйте их именно в таких случаях. Если вам нужны произвольные ключи, лучше подойдёт обычный объект.
